### 操作系统第三次作业

#### （1）读者写者问题（写者优先）

```c
int readers = 0;
Semaphore mutex = 1;
Semaphore roomEmpty = 1;
Semaphore writeBlock = 1; // 用于阻止新读者

Writer {
    P(writeBlock); // 阻止新读者
        P(roomEmpty);  // 确保写者独占
        	write; // critical region
        V(roomEmpty);
    V(writeBlock); // 允许等待的读者或写者进入
}

Reader {
    P(writeBlock); // 检查是否有写者等待
        P(mutex);      // 保护读者计数
            readers = readers + 1;
            if (readers == 1) {
                P(roomEmpty); // 第一个读者，阻止写者
            }
        V(mutex); 
    V(writeBlock); // 允许其他读者或写者检查
    
    read; // critical region
    
    P(mutex);      // 保护读者计数
        readers = readers - 1;
        if (readers == 0) {
            V(roomEmpty); // 最后一个读者，允许写者
        }
    V(mutex);
}

```

关键在于`wirteBlock`，确保了有写者排队之后，阻塞所有读者的进入（锁住了`readers`增加的过程）。

#### （2）寿司店问题

```c
int seatsOccupied = 0; // 当前已占用的座位数
Semaphore mutex = 1; // 用于保护对座位数的访问
Semaphore allSeatsEmpty = 1; // 用于等待所有座位都空出来

Customer {
    if (seatsOccupied < 5) {
        // 如果有空座位，顾客立即就坐
        P(mutex);
            seatsOccupied = seatsOccupied + 1;
            if (seatsOccupied == 5) {
                P(allSeatsEmpty);
            }
        V(mutex);
        eat; // critical region
        P(mutex);
            seatsOccupied = seatsOccupied - 1;
            if (seatsOccupied == 0) {
                V(allSeatsEmpty);
            }
        V(mutex);
    } else {
        // 如果没有空座位，顾客等待所有座位都空出来
        P(allSeatsEmpty);
            // 顾客就坐
            P(mutex);
                seatsOccupied = seatsOccupied + 1;
            V(mutex);
       	V(allSeatsEmpty);
        eat; // critical region
        P(mutex);
        	seatsOccupied = seatsOccupied - 1; // 这一组顾客离开，所有座位都空出来
        	if (seatsOccupied == 0) {
                V(allSeatsEmpty);
            }
        V(mutex);
    }
}

```

#### （3）进门问题

* P、V的含义：

  * P：检查信号量的值。如果信号量的值大于0，则将其减1，表示进程获得了资源。如果信号量的值为0，则进程被阻塞，直到信号量的值变为正数。
  * V：它会将信号量的值加1，表示释放了一个资源。如果此时有其他进程因为等待这个信号量而被阻塞，那么其中一个进程会被唤醒。

* ```c
  int employeesArrived = 0; // 已刷卡员工数
  int employeesEntered = 0; // 已进入员工数
  Semaphore allArrived = 0; // 所有员工都已刷卡
  Semaphore mutex = 1; // 保护对employeesArrived和employeesEntered的访问
  
  Employee {
      P(mutex);
          employeesArrived = employeesArrived + 1;
          if (employeesArrived == 5)
              V(allArrived); // 最后一个员工刷卡，唤醒所有等待的员工
      V(mutex);
      
      P(allArrived); // 等待所有员工都刷卡   
          enterOffice; // 进入公司
          P(mutex);
      		employeesEntered = employeesEntered + 1;
              if (employeesEntered == 5) {
                  closeDoor; // 最后一个员工负责关门
              }
      	V(mutex);
      V(allArrived); // 允许下一个员工通过大门

#### （4）搜索-插入-删除问题

```c
int searchNum = 0; 
Semaphore searchMutex = 1; // 控制搜索数量变量
Semaphore insertMutex = 1; // 控制插入操作的互斥
Semaphore deleteMutex = 1; // 控制删除操作的互斥
Semaphore searchDeleteMutex = 1; // 控制搜索和删除操作的互斥
Semaphore insertDeleteMutex = 1; // 控制插入和删除操作的互斥

// 搜索线程
Searcher {
    P(searchMutex);
    	searchNum = searchNum + 1;
    	if (searchNum == 1) {            
            P(searchDeleteMutex); // 阻止删除操作
        }  	
    V(searchMutex); 
    searchList;
    P(searchMutex);
    	searchNum = searchNum - 1;
    	if (searchNum == 0) {
            V(searchDeleteMutex);
        }  	
    V(searchMutex); 
}

// 插入线程
Inserter {
    P(insertMutex);
    	P(insertDeleteMutex); // 阻止删除操作
    		insertIntoList;
    	V(insertDeleteMutex);
    V(insertMutex);
}

// 删除线程
Deleter {
    P(deleteMutex);
    	P(searchDeleteMutex);
    		P(insertDeleteMutex);
            	deleteFromList;
    		V(insertDeleteMutex);
    	V(searchDeleteMutex);
    V(deleteMutex);
}
```

* 搜索线程：

  * 当有删除线程工作时，`searchDeleteMutex`被删除线程持有，无法进入。
  * 当无删除线程工作时，可以进入，维护`searchNum`。第一个搜索线程进入时，获取`searchDeleteMutex`，阻止删除线程；最后一个搜索线程离开时，释放`searchDeleteMutex`。
  
* 插入线程：

  * 当有删除线程工作时，`insertDeleteMutex`被删除线程持有，无法进入。
  * 当有其他插入线程工作时，`insertMutex`被其他插入线程持有，无法进入。
  * 当无删除线程工作且无其他插入线程工作时，可以进入，获取`insertDeleteMutex`和`insertMutex`，离开时释放。
  
* 删除线程：

  * 当有搜索线程工作时，`searchDeleteMutex`被持有，无法进入。
  * 当有插入线程工作时，`insertDeleteMutex`被持有，无法进入。
  * 当其他删除线程工作时，`deleteMutex`被持有，无法进入。
  * 当无其他线程工作时，可以进入，获取以上三个锁。
